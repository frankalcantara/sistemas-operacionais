1. Visão Geral do Projeto

"Este projeto implementa um sistema multi-threaded que realiza crawling de páginas da Wikipedia, calcula hashes dos textos extraídos e processa esses hashes através de um pipeline de threads especializadas, demonstrando conceitos avançados de programação concorrente."

2. Arquitetura do Sistema

Visão Geral da Arquitetura

Pipeline de Processamento
  1. Crawling: Threads fazem requisições HTTP para Wikipedia
  2. Buffer de Textos: Armazena conteúdo extraído (máx. 100 textos)
  3. Hash Production: Threads calculam SHA-1 dos textos
  4. Buffer de Hashes: Armazena hashes calculados (máx. 50 hashes)
  5. Processing: Threads consomem e validam hashes finais
Threads de Suporte
  • Monitor: Coleta métricas do sistema
  • Admin: Recebe comandos do dashboard
  • Logger: Registra eventos importantes
  • Cleanup: Limpeza periódica de recursos

5. Detalhes de Implementação

Função SHA-1 Simplificada

Os alunos devem implementar uma versão simplificada do algoritmo SHA-1:

class SimpleSHA1 {
private:
    // Valores iniciais de hash (constantes SHA-1)
    uint32_t h[5] = {
        0x67452301, 0xEFCDAB89, 0x98BADCFE, 
        0x10325476, 0xC3D2E1F0
    };
    
    // Função de rotação à esquerda
    uint32_t leftRotate(uint32_t value, unsigned int count) {
        return (value << count) | (value >> (32 - count));
    }
    
    // Funções auxiliares simplificadas
    uint32_t f(int t, uint32_t b, uint32_t c, uint32_t d) {
        if (t < 20) return (b & c) | ((~b) & d);
        else if (t < 40) return b ^ c ^ d;
        else if (t < 60) return (b & c) | (b & d) | (c & d);
        else return b ^ c ^ d;
    }
    
public:
    std::string hash(const std::string& message) {
        // Implementação simplificada com menos rounds
        // Padding do message
        std::string padded = message + "\x80";
        while ((padded.length() % 64) != 56) {
            padded += "\x00";
        }
        
        // Adicionar length (simplificado)
        uint64_t length = message.length() * 8;
        for (int i = 0; i < 8; i++) {
            padded += static_cast<char>(length >> (56 - i * 8));
        }
        
        // Processar blocos (implementação simplificada)
        for (size_t chunk = 0; chunk < padded.length(); chunk += 64) {
            processChunk(padded.substr(chunk, 64));
        }
        
        // Converter para string hexadecimal
        return toHexString();
    }
    
private:
    void processChunk(const std::string& chunk) {
        // Implementação dos 80 rounds simplificada para 40 rounds
        uint32_t w[80];
        
        // Preparar message schedule
        for (int i = 0; i < 16; i++) {
            w[i] = (chunk[i*4] << 24) | (chunk[i*4+1] << 16) | 
                   (chunk[i*4+2] << 8) | chunk[i*4+3];
        }
        
        for (int i = 16; i < 40; i++) {  // Reduzido de 80 para 40
            w[i] = leftRotate(w[i-3] ^ w[i-8] ^ w[i-14] ^ w[i-16], 1);
        }
        
        // Inicializar variáveis de trabalho
        uint32_t a = h[0], b = h[1], c = h[2], d = h[3], e = h[4];
        
        // Main loop (40 rounds ao invés de 80)
        for (int i = 0; i < 40; i++) {
            uint32_t temp = leftRotate(a, 5) + f(i, b, c, d) + e + w[i] + getK(i);
            e = d; d = c; c = leftRotate(b, 30); b = a; a = temp;
        }
        
        // Adicionar aos valores de hash
        h[0] += a; h[1] += b; h[2] += c; h[3] += d; h[4] += e;
    }
    
    uint32_t getK(int t) {
        if (t < 20) return 0x5A827999;
        else return 0x6ED9EBA1;  // Simplificado para 2 constantes
    }
    
    std::string toHexString() {
        std::stringstream ss;
        for (int i = 0; i < 5; i++) {
            ss << std::hex << std::setfill('0') << std::setw(8) << h[i];
        }
        return ss.str();
    }
};

Estruturas de Dados Thread-Safe

// Text Buffer Thread-Safe
template<typename T>
class ThreadSafeBuffer {
private:
    std::queue<T> buffer;
    mutable std::mutex mtx;
    std::condition_variable cv_not_full;
    std::condition_variable cv_not_empty;
    size_t max_size;
    std::atomic<bool> shutdown{false};
    
public:
    ThreadSafeBuffer(size_t max_sz) : max_size(max_sz) {}
    
    void push(const T& item) {
        std::unique_lock<std::mutex> lock(mtx);
        cv_not_full.wait(lock, [this] { 
            return buffer.size() < max_size || shutdown.load(); 
        });
        
        if (shutdown.load()) return;
        
        buffer.push(item);
        cv_not_empty.notify_one();
    }
    
    bool pop(T& item) {
        std::unique_lock<std::mutex> lock(mtx);
        cv_not_empty.wait(lock, [this] { 
            return !buffer.empty() || shutdown.load(); 
        });
        
        if (shutdown.load() && buffer.empty()) return false;
        
        item = buffer.front();
        buffer.pop();
        cv_not_full.notify_one();
        return true;
    }
    
    size_t size() const {
        std::lock_guard<std::mutex> lock(mtx);
        return buffer.size();
    }
    
    void shutdown_buffer() {
        shutdown.store(true);
        cv_not_full.notify_all();
        cv_not_empty.notify_all();
    }
};

Sistema de Métricas

struct SystemMetrics {
    // Contadores atômicos para thread safety
    std::atomic<int> texts_crawled{0};
    std::atomic<int> hashes_calculated{0};
    std::atomic<int> hashes_processed{0};
    
    // Métricas de performance
    std::atomic<double> avg_hash_time{0.0};
    std::atomic<double> avg_crawl_time{0.0};
    
    // Status dos buffers
    std::atomic<int> text_buffer_size{0};
    std::atomic<int> hash_buffer_size{0};
    
    // Timestamp de início
    std::chrono::steady__clock_::time_point start_time;
    
    // Método para calcular taxa de processamento
    double getProcessingRate() const {
        auto now = std::chrono::steady__clock_::now();
        auto duration = std::chrono::duration_cast<std::chrono::seconds>
                       (now - start_time).count();
        return duration > 0 ? static_cast<double>(texts_crawled.load()) / duration : 0.0;
    }
    
    // Serializar para JSON
    nlohmann::json toJson() const {
        nlohmann::json j;
        j["texts_crawled"] = texts_crawled.load();
        j["hashes_calculated"] = hashes_calculated.load();
        j["hashes_processed"] = hashes_processed.load();
        j["avg_hash_time"] = avg_hash_time.load();
        j["avg_crawl_time"] = avg_crawl_time.load();
        j["text_buffer_size"] = text_buffer_size.load();
        j["hash_buffer_size"] = hash_buffer_size.load();
        j["processing_rate"] = getProcessingRate();
        return j;
    }
};

Protocolo de Comunicação TCP

Backend C++ (Servidor TCP)

class DashboardServer {
private:
    SOCKET server_socket;
    std::thread server_thread;
    std::atomic<bool> running{true};
    SystemMetrics* metrics;
    
public:
    void start(int port = 8888) {
        WSADATA wsaData;
        WSAStartup(MAKEWORD(2,2), &wsaData);
        
        server_socket = socket(AF_INET, SOCK_STREAM, 0);
        
        sockaddr_in server_addr;
        server_addr.sin_family = AF_INET;
        server_addr.sin_addr.s_addr = INADDR_ANY;
        server_addr.sin_port = htons(port);
        
        bind(server_socket, (sockaddr*)&server_addr, 
             sizeof(server_addr));
        listen(server_socket, 5);
        
        server_thread = std::thread(&DashboardServer::accept_clients, this);
    }
    
private:
    void accept_clients() {
        while (running.load()) {
            sockaddr_in client_addr;
            int client_size = sizeof(client_addr);
            SOCKET client_socket = accept(server_socket, 
                                        (sockaddr*)&client_addr, 
                                        &client_size);
            
            if (client_socket != INVALID_SOCKET) {
                std::thread client_thread(&DashboardServer::handle_client, 
                                        this, client_socket);
                client_thread.detach();
            }
        }
    }
    
    void handle_client(SOCKET client_socket) {
        while (running.load()) {
            // Enviar métricas
            std::string metrics_json = metrics->toJson().dump();
            send(client_socket, metrics_json.c_str(), 
                 metrics_json.length(), 0);
            
            // Receber comandos
            char buffer[1024] = {0};
            int bytes = recv(client_socket, buffer, 1024, 0);
            if (bytes > 0) {
                processCommand(std::string(buffer, bytes));
            }
            
            std::this_thread::sleep_for(std::chrono::seconds(1));
        }
        closesocket(client_socket);
    }
    
    void processCommand(const std::string& json_str) {
        try {
            auto cmd = nlohmann::json::parse(json_str);
            std::string action = cmd["action"];
            
            if (action == "pause_crawlers") {
                // Pausar threads de crawler
            } else if (action == "adjust_speed") {
                double multiplier = cmd["value"];
                // Ajustar velocidade
            } else if (action == "clear_buffers") {
                // Limpar buffers
            }
        } catch (const std::exception& e) {
            // Log error
        }
    }
};

class BackendClient:
    def __init__(self, host='localhost', port=8888):
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.host = host
        self.port = port
        self.connected = False
        self.running = True
        
    def connect(self):
        try:
            self.socket.connect((self.host, self.port))
            self.connected = True
            
            # Iniciar thread de recepção
            self.recv_thread = threading.Thread(target=self.receive_loop)
            self.recv_thread.daemon = True
            self.recv_thread.start()
            
        except Exception as e:
            print(f"Erro de conexão: {e}")
    
    def receive_loop(self):
        while self.running and self.connected:
            try:
                data = self.socket.recv(1024)
                if data:
                    metrics = json.loads(data.decode())
                    self.update_dashboard(metrics)
                else:
                    break
            except Exception as e:
                print(f"Erro de recepção: {e}")
                break
    
    def send_command(self, command):
        if self.connected:
            try:
                json_data = json.dumps(command)
