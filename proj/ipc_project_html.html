1. Introdução à Comunicação Entre Processos

"A Comunicação entre Processos (IPC - Inter-Process Communication) é um conjunto de mecanismos que os sistemas operacionais fornecem para permitir que processos distintos gerenciem e sincronizem suas ações. Dominar esses conceitos é fundamental para o desenvolvimento de software robusto e eficiente."

Fundamentos de Sistemas Operacionais
Propósito do Projeto

Este projeto visa solidificar o entendimento teórico através da criação de uma ferramenta prática e visual que demonstre o funcionamento dos principais mecanismos de IPC, com uma arquitetura moderna que separa a lógica de backend da interface do usuário.

Desenvolver uma aplicação com backend em C++23/C que gerencie a comunicação entre processos utilizando três mecanismos distintos de IPC: Pipes Anônimos, Sockets (locais) e Memória Compartilhada. A lógica de C++ será visualizada e controlada por uma interface gráfica desenvolvida em Python ou HTML/JavaScript.

4. Mecanismos de IPC

Pipes Anônimos

Comunicação unidirecional entre processos relacionados (pai-filho). Simples e eficiente para dados pequenos.

// Exemplo básico de pipe
int pipefd[2];
if (pipe(pipefd) == -1) {
    perror("pipe");
    exit(1);
}

pid_t pid = fork();
if (pid == 0) {
    // Processo filho - escrever
    close(pipefd[0]);
    write(pipefd[1], "Hello", 5);
    close(pipefd[1]);
} else {
    // Processo pai - ler
    close(pipefd[1]);
    char buffer[100];
    read(pipefd[0], buffer, 100);
    close(pipefd[0]);
}

Sockets Locais

Comunicação bidirecional usando protocolo cliente-servidor. Mais flexível que pipes.

// Socket servidor local
int server_fd;
struct sockaddr_un server_addr;

server_fd = socket(AF_UNIX, SOCK_STREAM, 0);
server_addr.sun_family = AF_UNIX;
strcpy(server_addr.sun_path, "/tmp/ipc_socket");

bind(server_fd, 
     (struct sockaddr*)&server_addr, 
     sizeof(server_addr));
     
listen(server_fd, 5);

// Aceitar conexões
int client_fd = accept(server_fd, NULL, NULL);
char buffer[1024];
recv(client_fd, buffer, 1024, 0);

Memória Compartilhada

Acesso direto à mesma região de memória. Mais rápido mas requer sincronização cuidadosa.

// Criar segmento de memória compartilhada
key_t key = ftok("shared_mem_key", 65);
int shmid = shmget(key, 1024, 
                   0666 | IPC_CREAT);

// Anexar à memória
char *shared_memory = (char*) shmat(shmid, NULL, 0);

// Mutex para sincronização
pthread_mutex_t *mutex = 
    (pthread_mutex_t*)(shared_memory + 100);
    
pthread_mutex_lock(mutex);
strcpy(shared_memory, "Data");
pthread_mutex_unlock(mutex);

// Desanexar
shmdt(shared_memory);

Comparação dos Mecanismos IPC

Quando Usar Cada Mecanismo

Pipes
  • • Comunicação simples pai-filho
  • • Dados pequenos e sequenciais
  • • Prototipagem rápida
Sockets
  • • Processos independentes
  • • Comunicação estruturada
  • • Aplicações cliente-servidor
Shared Memory
  • • Alto volume de dados
  • • Performance crítica
  • • Controle fino de sincronização

Exemplo: Estrutura do Backend C++

// main.cpp - Estrutura principal
#include <iostream>
#include <string>
#include <nlohmann/json.hpp>
#include "IPCManager.h"

class IPCManager {
public:
    enum IPCType { PIPES, SOCKETS, SHARED_MEMORY };
    
private:
    IPCType current_type;
    std::unique_ptr<IPCBase> ipc_handler;
    
public:
    void setIPCType(IPCType type) {
        switch(type) {
            case PIPES:
                ipc_handler = std::make_unique<PipeHandler>();
                break;
            case SOCKETS:
                ipc_handler = std::make_unique<SocketHandler>();
                break;
            case SHARED_MEMORY:
                ipc_handler = std::make_unique<SharedMemHandler>();
                break;
        }
        current_type = type;
    }
    
    void sendMessage(const std::string& message) {
        if (ipc_handler) {
            IPCResult result = ipc_handler->send(message);
            outputJSON(result);
        }
    }
    
    void receiveMessage() {
        if (ipc_handler) {
            IPCResult result = ipc_handler->receive();
            outputJSON(result);
        }
    }
    
private:
    void outputJSON(const IPCResult& result) {
        nlohmann::json output;
        output["timestamp"] = getCurrentTimestamp();
        output["ipc_type"] = getIPCTypeName(current_type);
        output["status"] = result.success ? "success" : "error";
        output["message"] = result.message;
        output["data"] = result.data;
        output["buffer_state"] = result.buffer_info;
        
        std::cout << output.dump() << std::endl;
    }
};

Exemplo: Interface Base IPC

// IPCBase.h - Interface comum
class IPCBase {
public:
    struct IPCResult {
        bool success;
        std::string message;
        std::string data;
        nlohmann::json buffer_info;
        std::chrono::system__clock_::time_point timestamp;
    };
    
    virtual ~IPCBase() = default;
    virtual IPCResult initialize() = 0;
    virtual IPCResult send(const std::string& data) = 0;
    virtual IPCResult receive() = 0;
    virtual IPCResult cleanup() = 0;
    virtual nlohmann::json getStatus() = 0;
};

// PipeHandler.h - Implementação específica
class PipeHandler : public IPCBase {
private:
    int pipe_fd[2];
    pid_t child_pid;
    bool initialized = false;
    
public:
    IPCResult initialize() override {
        IPCResult result;
        
        if (pipe(pipe_fd) == -1) {
            result.success = false;
            result.message = "Failed to create pipe: " + std::string(strerror(errno));
            return result;
        }
        
        child_pid = fork();
        if (child_pid == -1) {
            result.success = false;
            result.message = "Failed to fork process: " + std::string(strerror(errno));
            return result;
        }
        
        if (child_pid == 0) {
            // Processo filho - consumer
            close(pipe_fd[1]); // Fechar write end
            childProcess();
            exit(0);
        } else {
            // Processo pai - producer
            close(pipe_fd[0]); // Fechar read end
            initialized = true;
            result.success = true;
            result.message = "Pipe initialized successfully";
        }
        
        return result;
    }
    
    IPCResult send(const std::string& data) override {
        IPCResult result;
        result.timestamp = std::chrono::system__clock_::now();
        
        if (!initialized) {
            result.success = false;
            result.message = "Pipe not initialized";
            return result;
        }
        
        ssize_t bytes_written = write(pipe_fd[1], data.c_str(), data.length());
        if (bytes_written == -1) {
            result.success = false;
            result.message = "Write failed: " + std::string(strerror(errno));
        } else {
            result.success = true;
            result.message = "Data sent successfully";
            result.data = data;
            result.buffer_info["bytes_written"] = bytes_written;
        }
        
        return result;
    }
    
private:
    void childProcess() {
        char buffer[1024];
        while (true) {
            ssize_t bytes_read = read(pipe_fd[0], buffer, sizeof(buffer) - 1);
            if (bytes_read > 0) {
                buffer[bytes_read] = '\0';
                
                // Output JSON para stdout do processo filho
                nlohmann::json output;
                output["type"] = "received";
                output["data"] = std::string(buffer);
                output["bytes"] = bytes_read;
                output["timestamp"] = std::chrono::duration_cast<std::chrono::milliseconds>
                    (std::chrono::system__clock_::now().time_since_epoch()).count();
                
                std::cout << output.dump() << std::endl;
            } else if (bytes_read == 0) {
                break; // Pipe closed
            }
        }
    }
};

Exemplo: Estrutura de Testes

// test_pipes.cpp - Testes unitários com Google Test
#include <gtest/gtest.h>
#include "PipeHandler.h"

class PipeHandlerTest : public ::testing::Test {
protected:
    void SetUp() override {
        handler = std::make_unique<PipeHandler>();
    }
    
    void TearDown() override {
        if (handler) {
            handler->cleanup();
        }
    }
    
    std::unique_ptr<PipeHandler> handler;
};

TEST_F(PipeHandlerTest, InitializationSuccess) {
    IPCBase::IPCResult result = handler->initialize();
    EXPECT_TRUE(result.success);
    EXPECT_FALSE(result.message.empty());
}

TEST_F(PipeHandlerTest, SendReceiveMessage) {
    // Inicializar
    ASSERT_TRUE(handler->initialize().success);
    
    // Enviar mensagem
    std::string test_message = "Hello IPC World!";
    IPCBase::IPCResult send_result = handler->send(test_message);
    
    EXPECT_TRUE(send_result.success);
    EXPECT_EQ(send_result.data, test_message);
    EXPECT_GT(send_result.buffer_info["bytes_written"], 0);
}

TEST_F(PipeHandlerTest, ErrorHandling) {
    // Tentar enviar sem inicializar
    IPCBase::IPCResult result = handler->send("test");
    EXPECT_FALSE(result.success);
    EXPECT_THAT(result.message, ::testing::HasSubstr("not initialized"));
}

// Executar testes
int main(int argc, char **argv) {
    ::testing::InitGoogleTest(&argc, argv);
    return RUN_ALL_TESTS();
}

Exemplo: Interface Python com Tkinter

import tkinter as tk
from tkinter import ttk, scrolledtext
import subprocess
import json
import threading
import queue
from datetime import datetime

class IPCVisualizer:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("IPC Demonstration Tool")
        self.root.geometry("800x600")
        
        # Queue para comunicação entre threads
        self.message_queue = queue.Queue()
        
        # Processo backend atual
        self.backend_process = None
        self.current_ipc_type = "pipes"
        
        self.setup_ui()
        self.start_message_processor()
    
    def setup_ui(self):
        # Frame principal
        main_frame = ttk.Frame(self.root, padding="10")
        main_frame.grid(row=0, column=0, sticky=(tk.W, tk.E, tk.N, tk.S))
        
        # Seleção de IPC
        ipc_frame = ttk.LabelFrame(main_frame, text="IPC Mechanism Selection", padding="10")
        ipc_frame.grid(row=0, column=0, columnspan=2, sticky=(tk.W, tk.E), pady=(0, 10))
        
        self.ipc_var = tk.StringVar(value="pipes")
        ttk.Radiobutton(ipc_frame, text="Pipes Anônimos", variable=self.ipc_var, 
                       value="pipes", command=self.on_ipc_change).grid(row=0, column=0, padx=5)
        ttk.Radiobutton(ipc_frame, text="Sockets Locais", variable=self.ipc_var, 
                       value="sockets", command=self.on_ipc_change).grid(row=0, column=1, padx=5)
        ttk.Radiobutton(ipc_frame, text="Memória Compartilhada", variable=self.ipc_var, 
                       value="shared_mem", command=self.on_ipc_change).grid(row=0, column=2, padx=5)
        
        # Controles
        control_frame = ttk.LabelFrame(main_frame, text="Process Control", padding="10")
        control_frame.grid(row=1, column=0, columnspan=2, sticky=(tk.W, tk.E), pady=(0, 10))
        
        self.start_button = ttk.Button(control_frame, text="Start IPC", command=self.start_ipc)
        self.start_button.grid(row=0, column=0, padx=5)
        
        self.stop_button = ttk.Button(control_frame, text="Stop IPC", command=self.stop_ipc, state=tk.DISABLED)
        self.stop_button.grid(row=0, column=1, padx=5)
        
        # Status indicator
        self.status_label = ttk.Label(control_frame, text="Status: Stopped", foreground="red")
        self.status_label.grid(row=0, column=2, padx=20)
        
        # Input de mensagem
        input_frame = ttk.LabelFrame(main_frame, text="Send Message", padding="10")
        input_frame.grid(row=2, column=0, columnspan=2, sticky=(tk.W, tk.E), pady=(0, 10))
        
        self.message_entry = ttk.Entry(input_frame, width=50)
        self.message_entry.grid(row=0, column=0, padx=(0, 10))
        self.message_entry.bind("", self.send_message)
        
        self.send_button = ttk.Button(input_frame, text="Send", command=self.send_message, state=tk.DISABLED)
        self.send_button.grid(row=0, column=1)
        
        # Log de eventos
        log_frame = ttk.LabelFrame(main_frame, text="Event Log", padding="10")
        log_frame.grid(row=3, column=0, sticky=(tk.W, tk.E, tk.N, tk.S), pady=(0, 10))

    def on_ipc_change(self):
        self.current_ipc_type = self.ipc_var.get()
        self.log_message(f"IPC type changed to: {self.current_ipc_type.upper()}")
    
    def start_ipc(self):
        try:
            # Executar backend C++
            cmd = [f"./backend_{self.current_ipc_type}", "--json-output"]
            self.backend_process = subprocess.Popen(
                cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE,
                text=True, bufsize=1, universal_newlines=True
            )
            
            # Thread para ler output do backend
            self.reader_thread = threading.Thread(
                target=self.read_backend_output, daemon=True)
            self.reader_thread.start()
            
            # Atualizar UI
            self.start_button.config(state=tk.DISABLED)
            self.stop_button.config(state=tk.NORMAL)
            self.send_button.config(state=tk.NORMAL)
            self.status_label.config(text="Status: Running", foreground="green")
            
            self.log_message(f"Started {self.current_ipc_type.upper()} backend process")
            
        except Exception as e:
            self.log_message(f"Error starting backend: {str(e)}")
    
    def stop_ipc(self):
        if self.backend_process:
            self.backend_process.terminate()
            self.backend_process.wait()
            self.backend_process = None
        
        # Atualizar UI
        self.start_button.config(state=tk.NORMAL)
        self.stop_button.config(state=tk.DISABLED)
        self.send_button.config(state=tk.DISABLED)
        self.status_label.config(text="Status: Stopped", foreground="red")
        
        self.log_message("Backend process stopped")
    
    def send_message(self, event=None):
        message = self.message_entry.get().strip()
        if not message:
            return
        
        try:
            if self.backend_process:
                # Enviar comando JSON para o backend
                command = {
                    "action": "send_message",
                    "data": message,
                    "timestamp": datetime.now().isoformat()
                }
                self.backend_process.stdin.write(json.dumps(command) + "\n")
                self.backend_process.stdin.flush()
                
                self.log_message(f"Sent: {message}")
                self.message_entry.delete(0, tk.END)
        
        except Exception as e:
            self.log_message(f"Error sending message: {str(e)}")
    
    def read_backend_output(self):
        try:
            for line in self.backend_process.stdout:
                line = line.strip()
                if line:
                    self.message_queue.put(('output', line))
        except Exception as e:
            self.message_queue.put(('error', f"Backend reading error: {str(e)}"))
    
    def start_message_processor(self):
        def process_messages():
            try:
                while True:
                    msg_type, data = self.message_queue.get(timeout=0.1)
                    
                    if msg_type == 'output':
                        try:
                            json_data = json.loads(data)
                            self.process_backend_message(json_data)
                        except json.JSONDecodeError:
                            self.log_message(f"Raw output: {data}")
                    
                    elif msg_type == 'error':
                        self.log_message(data)
                        
            except queue.Empty:
                pass
            
            # Reagendar para próxima verificação
            self.root.after(100, process_messages)
        
        process_messages()
    
    def process_backend_message(self, data):
        timestamp = data.get('timestamp', datetime.now().isoformat())
        ipc_type = data.get('ipc_type', 'unknown')
        status = data.get('status', 'unknown')
        message = data.get('message', '')
        
        # Log de eventos
        log_entry = f"[{timestamp[:19]}] {ipc_type.upper()}: {message}"
        if status != 'success':
            log_entry += f" (Status: {status})"
        
        self.log_message(log_entry)
        
        # Atualizar status visual
        if 'buffer_state' in data:
            self.update_status_display(data['buffer_state'])
    
    def update_status_display(self, buffer_state):
        self.status_text.delete('1.0', tk.END)
        
        status_info = f"Buffer Status:\n"
        status_info += f"Type: {self.current_ipc_type.upper()}\n"
        status_info += f"State: {json.dumps(buffer_state, indent=2)}\n"
        
        if self.current_ipc_type == 'pipes':
            status_info += f"\nPipe Info:\n"
            status_info += f"• Read FD: {buffer_state.get('read_fd', 'N/A')}\n"
            status_info += f"• Write FD: {buffer_state.get('write_fd', 'N/A')}\n"
            status_info += f"• Bytes Written: {buffer_state.get('bytes_written', 0)}\n"
        
        elif self.current_ipc_type == 'sockets':
            status_info += f"\nSocket Info:\n"
            status_info += f"• Server Status: {buffer_state.get('server_status', 'Unknown')}\n"
            status_info += f"• Connections: {buffer_state.get('active_connections', 0)}\n"
            status_info += f"• Messages Queued: {buffer_state.get('message_queue_size', 0)}\n"
        
        elif self.current_ipc_type == 'shared_mem':
            status_info += f"\nShared Memory Info:\n"
            status_info += f"• Memory ID: {buffer_state.get('shmem_id', 'N/A')}\n"
            status_info += f"• Size: {buffer_state.get('size', 0)} bytes\n"
            status_info += f"• Mutex Status: {buffer_state.get('mutex_status', 'Unknown')}\n"
        
        self.status_text.insert('1.0', status_info)
    
    def log_message(self, message):
        timestamp = datetime.now().strftime("%H:%M:%S")
        log_entry = f"[{timestamp}] {message}\n"
        
        self.log_text.insert(tk.END, log_entry)
        self.log_text.see(tk.END)
    
    def run(self):
        self.root.protocol("WM_DELETE_WINDOW", self.on_closing)
        self.root.mainloop()
    
    def on_closing(self):
        if self.backend_process:
            self.stop_ipc()
        self.root.destroy()

if __name__ == "__main__":
    app = IPCVisualizer()
    app.run()
